Quick Start

import OpenAI from 'openai';

const client = new OpenAI({
  baseURL: 'https://core-ai-proxy.workers.dev/api/v1',
  apiKey: 'your-key'
});

const response = await client.chat.completions.create({
  model: 'gpt-4', // or claude-3-5-sonnet-20240620, gemini-1.5-pro, @cf/meta/llama-3-8b-instruct
  messages: [{ role: 'user', content: 'Hello!' }]
});
